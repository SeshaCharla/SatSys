\section*{Nomenclature}
\textbf{Symbols:}
\begin{itemize}[leftmargin=3.5em]
    \item[$\{\bullet\}$] Number of moles of $\bullet$. $(moles)$
    \item[$\lbrack\bullet\rbrack$] Concentration of $\bullet$. $(mol/m^3)$
    \item[$A_{scr}$] Pre-exponential factor of SCR reaction
    \item[$E_{scr}$] Activation Energy of SCR reaction
    \item[$F$] Mass flow rate of the exhaust gases $(g/s)$
    \item[$I(\theta)$] Fisher Information Matrix
    \item[$k_i$] Coefficients of polynomial temperature model for $k_{scr}$
    \item[$N_{sat}$] Number of data points close to catalyst saturation
    \item[$N$] Total number of data points
    \item[$N_{sat}^{min}$] Minimum number of data points close to catalyst saturation for reliable parameter estimation
    \item[$r_i$] Coefficients of polynomial temperature model for $\Gamma$
    \item[$R$] Universal gas constant
    \item[$S_1, S_2$] Coefficients of the temperature model for $\Gamma$, $S_1, S_2 \geq 0$
    \item[$T$] Temperature
    \item[$T_w$] Wald test statistic for catalyst aging detection
    \item[$u_1$] Engine-out (SCR-in) $NO_x$ concentration
    \item[$V_{scr}$] Volume of the catalyst chamber
    \item[$x_1$] Tailpipe (SCR-out) $NO_x$ concentration
    \item[$y_1$] Tailpipe (SCR-out) $NO_x$ sensor measurement
    \item[$\alpha_{sat}$] Flow and inlet $NO_x$ normalized $NO_x$ reduction under catalyst saturation
    \item[$\beta$] Threshold for Wald test statistic for catalyst aging detection
    \item[$\gamma(k)$] Average surface concentration of adsorbed ammonia in the sample [k] to [k+1]
    \item[$\Gamma$] Surface concentration of the total storage capacity for the catalyst $(mol/m^2)$
    \item[$\varepsilon_\chi$] Error in $NO_x$ due to ammonia cross-sensitivity
    \item[$\varepsilon_\eta$] Model structure error between saturated catalyst model and actual system
    \item[$\eta_{sat}$] $NO_x$ reduction per time-step under catalyst saturation
    \item[$\phi_{sat}$] Regression vector for saturated catalyst model
    \item[$\rho$] Density of the exhaust gases
    \item[$\sigma_{\eta}$] Standard deviation in $NO_x$
    \item[$\sigma$] Scale parameter for half-normal distribution
    \item[$\theta_{sat}$] Parameter vector for saturated catalyst model
    \item[$\theta_{sat}^{dg}$] Parameter estimates for degreened catalyst
    \item[$\chi$] Cross-sensitivity factor
\end{itemize}
\textbf{Abbreviations:}
\begin{itemize}[leftmargin=3.5em]
    \item[ASC] Ammonia Slip Catalyst
    \item[CSTR] Continuously Stirred Tank Reactor
    \item[DG] Degreened
    \item[FTP] Federal Test Procedure
    \item[MLE] Maximum Likelihood Estimate
    \item[PDF] Probability Density Function
    \item[PFR] Plug Flow Reactor
    \item[RMC] Ramped Mode Cycle
    \item[SCR] Selective Catalytic Reduction
\end{itemize}
